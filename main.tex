	\fcolorbox{boxcol}{boxcol}{\begin{minipage}{\columnwidth}{\sffamily\LARGE\color{white}#1 }\end{minipage}}	%Draws solid colour box around title

	\vspace{0.5cm}						%Add spacing below box
}
